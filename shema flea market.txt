CREATE SCHEMA flea_market;

SET SEARCH_PATH = flea_market;

CREATE TABLE person_role (
  id   SERIAL PRIMARY KEY,
  role VARCHAR(124) UNIQUE NOT NULL
);

CREATE TABLE person (
  id         SERIAL PRIMARY KEY,
  avatar     VARCHAR(1024),
  login      VARCHAR(124) UNIQUE NOT NULL,
  first_name VARCHAR(124)        NOT NULL,
  last_name  VARCHAR(124)        NOT NULL,
  age        SMALLINT            NOT NULL,
  mail       VARCHAR(124) UNIQUE NOT NULL,
  password   VARCHAR(124)        NOT NULL,
  role       INTEGER REFERENCES person_role (id) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE category (
  id            SERIAL PRIMARY KEY,
  category_name VARCHAR(124) UNIQUE NOT NULL,
  foto          VARCHAR(124)
);

CREATE TABLE heading (
  id           SERIAL PRIMARY KEY,
  heading_name VARCHAR(124) NOT NULL,
  category_id  INTEGER REFERENCES category (id)ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE resource (
  id            SERIAL PRIMARY KEY,
  resource_name VARCHAR(124) NOT NULL,
  foto          VARCHAR(1024),
  heading_id    INTEGER REFERENCES heading (id) ON DELETE CASCADE ON UPDATE CASCADE,
  category_id   INTEGER REFERENCES category (id) ON DELETE CASCADE ON UPDATE CASCADE,
  author        VARCHAR(124) REFERENCES person (login) ON DELETE CASCADE ON UPDATE CASCADE,
  price         INTEGER,
  text          VARCHAR(1024) NOT NULL
);

DROP TABLE resource;

CREATE TABLE resource_person (
  resources_id INTEGER REFERENCES resource (id) ON DELETE CASCADE ON UPDATE CASCADE,
  person_id    INTEGER REFERENCES person (id) ON DELETE CASCADE ON UPDATE CASCADE,
  PRIMARY KEY (resources_id, person_id)
);

CREATE TABLE store_basket (
  foto   VARCHAR(124) NOT NULL ,
  person_id    INTEGER REFERENCES person (id) ON DELETE CASCADE ON UPDATE CASCADE,
  resources_id INTEGER REFERENCES resource (id) ON DELETE CASCADE ON UPDATE CASCADE,
  PRIMARY KEY (person_id,resources_id)
);

DROP TABLE store_basket;

CREATE TABLE comment (
  id          SERIAL PRIMARY KEY,
  person_id    INTEGER REFERENCES person (id) ON DELETE CASCADE ON UPDATE CASCADE,
  resource_id INTEGER REFERENCES resource (id) ON DELETE CASCADE ON UPDATE CASCADE,
  text        VARCHAR(500)
);
